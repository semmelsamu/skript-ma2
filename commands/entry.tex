\newcommand{\entrysource}[3]{
    \stepcounter{subsection}
    \subsection*{\large \textsc{#1} \arabic{chapter}.\arabic{section}.\arabic{subsection} (#2)}
    
    \begin{adjustwidth}{3em}{0pt}
        \input{entries/#3}
    \end{adjustwidth}
}

\NewDocumentCommand{\entry}{ m m o }{
    \IfNoValueTF{#3}{
        \entrysource{#1}{#2}{#2}
    }{
        \entrysource{#1}{#2}{#3}
    }
}
