Eine Menge $K$ mit zwei Verknüpfungen, einer Addition und einer Multiplikation, 
\begin{align*}
    + : K \times K & \longrightarrow K \hspace{1cm} &
    \cdot : K \times K & \longrightarrow K
    \\
    (x, y) & \longmapsto x + y &
    (x, y) & \longmapsto x \cdot y = xy
\end{align*}
heißt \term{kommutativer Körper}, falls gilt:
\begin{enumerate}[label=(K\arabic*)]
    \item \term{Assoziativität}: $\forall x, y, z \in K :$\\$(x+y)+z = x+(y+z),$\\$(x \cdot y) \cdot z = x \cdot (y \cdot z)$
    \item \term{Neutrales Element (Null)}: $\exists 0 \in K : \forall x \in K :$\\$x+0=x=0+x$
    \item \term{Neutrales Element (Eins)}: $\exists 1 \in K \backslash \{0\} : \forall x \in R :$\\$x \cdot 1=x=1 \cdot x$
    \item \term{Inverse Elemente (Negative)}: $\forall x \in R : \exists -x \in K :$\\$x + (-x) = 0 = (-x) + x$
    \item \term{Inverse Elemente (Inverse)}: $\forall x \in K \backslash \{0\} : \exists x^{-1} \in K :$\\$x \cdot x^{-1} = 1 = x^{-1} \cdot x$
    \item \term{Kommutativität}: $\forall x, y \in K :$\\$x+y=y+x, x \cdot y = y \cdot x$
    \item \term{Distributivität}: $\forall x, y, z \in K :$\\$x \cdot (y+z) = (x \cdot y) + (x \cdot z),$\\$(x+y) \cdot z = (x \cdot z) + (y \cdot z)$
\end{enumerate}
