Eine Menge $K$ mit zwei Verknüpfungen, einer \term{Addition} und einer \term{Multiplikation}, 
\begin{align*}
    + : K \times K & \longrightarrow K \hspace{1cm} &
    \cdot : K \times K & \longrightarrow K
    \\
    (x, y) & \longmapsto x + y &
    (x, y) & \longmapsto x \cdot y = xy
\end{align*}
heißt \term{Körper}, falls gilt:
\begin{description}
    \item[\term{Assoziativität}] $\forall x, y, z \in K : (x+y)+z = x+(y+z), \ (x \cdot y) \cdot z = x \cdot (y \cdot z)$
    \item[\term{Neutrales Element}]
        \begin{description}[leftmargin=8em]
            \item[(Null)] $\exists 0 \in K : \forall x \in K : x+0=x=0+x$
            \item[(Eins)] $\exists 1 \in K \setminus \{0\} : \forall x \in R : x \cdot 1=x=1 \cdot x$
        \end{description}
    \item[\term{Inverse Elemente}] 
        \begin{description}
            \item[(Negative)] $\forall x \in R : \exists -x \in K : x + (-x) = 0 = (-x) + x$
            \item[(Inverse)] $\forall x \in K \setminus \{0\} : \exists x^{-1} \in K : x \cdot x^{-1} = 1 = x^{-1} \cdot x$
        \end{description}
    \item[\term{Kommutativität}] $\forall x, y \in K : x+y=y+x, \ x \cdot y = y \cdot x$
    \item[\term{Distributivität}] $\forall x, y, z \in K : x \cdot (y+z) = (x \cdot y) + (x \cdot z), \ (x+y) \cdot z = (x \cdot z) + (y \cdot z)$
\end{description}
