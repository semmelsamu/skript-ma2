\begin{enumerate}[label=(\arabic*)]

    \item $\lim (\lambda \cdot a_n) = \lambda \cdot \lim a_n$ \qquad f端r alle $\lambda \in \mathbb{C}$
    \item $\lim(a_n \pm b_n) = (\lim a_n) \pm (\lim b_n)$
    \item $\lim(a_n \cdot b_n) = (\lim a_n) \cdot (\lim b_n)$
    \item Ist $\lim b_n \neq 0$, so existiert ein $n_0 \in \N$ mit $b_n \neq 0$ f端r alle $n \geq n_0$ und es gilt:
    $$\lim_{\substack{n \to \infty \\ n \geq n_0}} \frac{a_n}{b_n} = \frac{\lim a_n}{\lim b_n}$$

    $$\lim \sqrt[k]{a_n} = \sqrt[k]{\lim a_n} \qquad \text{f端r alle $k \in \N$}$$
\end{enumerate}
