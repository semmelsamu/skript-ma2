Unter den Hyperbolischen Funktionen bzw. \term{Hyperbelfunktionen} versteht man die folgenden Funktionen:
\begin{itemize}[label="",leftmargin=0pt]
    \item \term{Hyperbelsinus} \, $\sinh : \R \to \R, \quad \sinh(x) := \frac{1}{2}(e^x-e^{-x})$
    \item \term{Hyperbelkosinus} \, $\cosh : \R \to [1, \infty), \quad \cosh(x) := \frac{1}{2}(e^x+e^{-x})$
    \item \term{Hyperbeltangens} \, $\tanh : \R \to (-1, 1), \quad \tanh(x) := \frac{\sinh(x)}{\cosh(x)}$
    \item \term{Hyperbelkotangens} \, $\coth : \R \setminus \{0\} \to \R \setminus [-1, 1], \quad \coth(x) := \frac{1}{\tanh(x)}$
\end{itemize}
