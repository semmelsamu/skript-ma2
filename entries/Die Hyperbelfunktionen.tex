Unter den Hyperbolischen Funktionen bzw. \term{Hyperbelfunktionen} versteht man die folgenden Funktionen:
\begin{itemize}[label={},leftmargin=*]
    \item \term{Hyperbelsinus} $\quad \sinh : \R \to \R, \quad \sinh(x) := \frac{1}{2}(e^x-e^{-x})$
    \item \term{Hyperbelkosinus} $\quad \cosh : \R \to [1, \infty), \quad \cosh(x) := \frac{1}{2}(e^x+e^{-x})$
    \item \term{Hyperbeltangens} $\quad \tanh : \R \to (-1, 1), \quad \tanh(x) := \frac{\sinh(x)}{\cosh(x)}$
    \item \term{Hyperbelkotangens} $\quad \coth : \R \setminus \{0\} \to \R \setminus [-1, 1], \quad \coth(x) := \frac{1}{\tanh(x)}$
\end{itemize}
