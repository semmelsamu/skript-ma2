\begin{enumerate}[leftmargin=*]
    \item Die Funktion $f$ sei auf $[p, p + \eta)$ (für ein $\eta > 0$) definiert. Dann heißt $f$ an der Stelle $p$ \term{rechtsseitig stetig}, falls gilt:
    $$\lim_{x \to p+} f(x) = f(p)$$
    \item Die Funktion $f$ sei auf $(p - \eta, p]$ (für ein $\eta > 0$) definiert. Dann heißt $f$ an der Stelle $p$ \term{linksseitig stetig}, falls gilt:
    $$\lim_{x \to p-} f(x) = f(p)$$
\end{enumerate}
