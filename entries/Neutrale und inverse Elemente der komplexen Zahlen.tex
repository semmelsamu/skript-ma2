Aus den Rechenoperationen auf $\mathbb{C}$ folgen die neutralen und inversen Elemente:

\begin{itemize}[label={},leftmargin=*]

    \item \term{Neutrales Element der Addition}
    $$(0, 0)$$
    
    \item \term{Additives Inverses} zu $z = (a, b)$
    $$(-a, -b)$$
    
    \item \term{Neutrales Element der Multiplikation}
    $$(1, 0)$$
    
    \item \term{Multiplikatives Inverses} zu $(a, b) \neq (0, 0)$
    $$(a, b)^{-1} := \left(\frac{a}{a^2+b^2}, \frac{-b}{a^2+b^2}\right)$$
    
\end{itemize}
